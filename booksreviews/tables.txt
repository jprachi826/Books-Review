create table  books 
( bookid  int primary key,
  title   varchar(50)  not null,
  authors  varchar(100),
  publisher varchar(20),
  price   money,
  nopages int ,
  pubdate datetime,
  subcode  varchar(10)  references subjects(subcode),
  addedon  datetime
)
